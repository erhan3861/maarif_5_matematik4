            display: flex;

        /* Arazi karesi: CSS Grid ile bölmelere ayrılır */

    <h2>🌍 Adil Paylaşım</h2>

        <!-- Sol > Sağ, Eşit, Sol < Sağ seçimleri -->
        <div class="butonlar">
            <button class="karsilastir btn-sol" onclick="cevapla('sol')">&gt;</button>
            <button class="karsilastir btn-esit" onclick="cevapla('esit')">=</button>
            <button class="karsilastir btn-sag" onclick="cevapla('sag')">&lt;</button>
        </div>

        <div class="mesaj" id="mesaj"></div>

    <script>
    // ============================================================
    // 📚 BÖLÜM 1: SEVİYE VERİLERİ
    // ============================================================

    // Her nesne bir kesir karşılaştırma sorusunu temsil eder:
    //   sol  = Sol arazinin bilgileri
    //     pay    = Kesrin payı (sahip olunan parça sayısı)
    //     payda  = Kesrin paydası (toplam parça sayısı)
    //     sutun  = Grid'deki sütun sayısı
    //     satir  = Grid'deki satır sayısı
    //     deger  = Ondalık değeri (karşılaştırmak için)
    //
    // 💡 ÖNEMLİ: sutun × satir = payda olmalı!
    //   1/2: 1 sütun × 2 satır = 2 parça (payda)
    //   1/4: 2 sütun × 2 satır = 4 parça (payda)
    const seviyeler = [
        {
            sol: { pay:1, payda:2, sutun:1, satir:2, deger:0.5 },
            sag: { pay:1, payda:4, sutun:2, satir:2, deger:0.25 }
        }, // 1/2 > 1/4
        {
            sol: { pay:2, payda:4, sutun:2, satir:2, deger:0.5 },
            sag: { pay:1, payda:2, sutun:1, satir:2, deger:0.5 }
        }, // 2/4 = 1/2 (Denk Kesir!)
        {
            sol: { pay:3, payda:4, sutun:2, satir:2, deger:0.75 },
            sag: { pay:2, payda:3, sutun:1, satir:3, deger:0.66 }
        }  // 3/4 > 2/3
    ];

    // ============================================================
    // 📚 BÖLÜM 2: DEĞİŞKENLER
    // ============================================================

    let mevcutIdx = 0;

    window.onload = function() { seviyeYukle(); };

    // ============================================================
    // 📚 BÖLÜM 3: SEVİYE YÜKLEME
    // ============================================================

    function seviyeYukle() {
        if (mevcutIdx >= seviyeler.length) {
            document.querySelector('.tahta').innerHTML =
                '<h2 style="color:#33691e;">🏆 ADİL PAYLAŞIM!</h2>' +
                '<p>Tüm kadastro işlemlerini doğru tamamladın!</p>';
            return;
        }
        const s = seviyeler[mevcutIdx];
        // Template literal ile kesri göster
        document.getElementById('kesir-sol').innerText =
            `${s.sol.pay}/${s.sol.payda}`;
        document.getElementById('kesir-sag').innerText =
            `${s.sag.pay}/${s.sag.payda}`;
        document.getElementById('mesaj').innerText = '';
        araziCiz('arazi-sol', s.sol);
        araziCiz('arazi-sag', s.sag);
    }

    // ============================================================
    // 📚 BÖLÜM 4: ARAZİ ÇİZME
    // ============================================================

    function araziCiz(elemId, arazi) {
        const kutu = document.getElementById(elemId);
        kutu.innerHTML = '';   // Eski içeriği sil

        // CSS Grid: Kaç sütun? (repeat = tekrar et)
        kutu.style.gridTemplateColumns = `repeat(${arazi.sutun}, 1fr)`;
        kutu.style.gridTemplateRows    = `repeat(${arazi.satir}, 1fr)`;

        // payda kadar hücre oluştur
        // i < pay → Yeşil (sahipli), sonrası → Gri (sahipsiz)
        for (let i = 0; i < arazi.payda; i++) {
            const hucre = document.createElement('div');
            hucre.className = 'hucre';
            if (i < arazi.pay) {
                hucre.classList.add('sahip');  // classList.add = class ekle
            }
            kutu.appendChild(hucre);
        }
    }

    // ============================================================
    // 📚 BÖLÜM 5: CEVAP KONTROLÜ
    // ============================================================

    function cevapla(secilen) {
        const s = seviyeler[mevcutIdx];
        let dogru = '';
        if      (s.sol.deger > s.sag.deger) dogru = 'sol';
        else if (s.sag.deger > s.sol.deger) dogru = 'sag';
        else                                 dogru = 'esit';

        const mesaj = document.getElementById('mesaj');
        if (secilen === dogru) {
            mesaj.innerText = '✅ Doğru! Araziler doğru karşılaştırıldı.';
            mesaj.style.color = '#2e7d32';
            setTimeout(function() { mevcutIdx++; seviyeYukle(); }, 1800);
        } else {
            alert('Hatalı! Yeşil kareleri say ve karşılaştır.');
        }
    }
    </script>

<!-- ============================================================
🎯 GÖREVLER (Kodu değiştirerek dene!)
============================================================

GÖREV 1 (Kolay): Arazi rengini değiştir
→ .hucre.sahip { background: #8bc34a; } satırını bul
→ #8bc34a (yeşil) yerine #ff9f43 (turuncu) yaz
→ Sahipli arazi turuncu renk alacak!

GÖREV 2 (Kolay): Arazi çerçevesini kalınlaştır
→ .arazi { border: 4px solid #333; } satırını bul
→ 4px yerine 8px yaz
→ Arazi sınırı çok daha belirgin görünecek!

GÖREV 3 (Orta): Yeni bir karşılaştırma sorusu ekle
→ const seviyeler = [ ... ]; dizisine ekle:
   , {
       sol: { pay:1, payda:3, sutun:1, satir:3, deger:0.33 },
       sag: { pay:2, payda:6, sutun:2, satir:3, deger:0.33 }
   }
→ 1/3 ve 2/6 denk kesirdir! (Her ikisi de 0.33 = eşit)

GÖREV 4 (Zor): Kalan parça sayısını göster
→ araziCiz() fonksiyonunun içine, for döngüsünden SONRA ekle:
   const bosParca = arazi.payda - arazi.pay;
   const bilgi = document.createElement('div');
   bilgi.style.cssText = 'font-size:0.75rem; color:#888; margin-top:5px;';
   bilgi.innerText = bosParca + ' parça sahipsiz kaldı';
   kutu.parentElement.appendChild(bilgi);
→ Her arazinin altında "X parça sahipsiz kaldı" yazar!
