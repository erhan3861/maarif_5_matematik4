    <script>
    // ============================================================
    // 📚 BÖLÜM 1: SORU VERİLERİ
    // ============================================================

    // Her nesne bir soruyu temsil eder:
    //   yazi   = Kesirim ekranda görünen hali
    //   hedef  = Hangi referans noktasına en yakın? (0 / 0.5 / 1)
    //   ipucu  = Yanlış cevapta gösterilecek ipucu mesajı
    //
    // 💡 REFERANS NOKTASI MANTIĞI:
    //   1/12 ≈ 0.083 → 0'a yakın (çünkü 0'dan uzaklığı: 0.083)
    //   6/10 = 0.6   → 1/2'ye yakın (çünkü 0.5'e uzaklığı: 0.1)
    //   9/10 = 0.9   → 1'e yakın (çünkü 1'e uzaklığı: 0.1)
    const sorular = [
        { yazi: "1/12", hedef: 0,   ipucu: "12 parçadan sadece 1'i. Sıfıra çok yakın!" },
        { yazi: "6/10", hedef: 0.5, ipucu: "Yarısı 5/10 eder. 6/10, yarıma çok yakın!" },
        { yazi: "9/10", hedef: 1,   ipucu: "10/10 tam olurdu. 9/10 buna çok yakın!" },
        { yazi: "1/8",  hedef: 0,   ipucu: "8 parçadan sadece 1'i = 0.125. Sıfıra çok yakın!" }
    ];

    // ============================================================
    // 📚 BÖLÜM 2: DEĞİŞKENLER
    // ============================================================

    let mevcutIdx = 0;   // Hangi soruda?

    window.onload = function() { soruYukle(); };

    // ============================================================
    // 📚 BÖLÜM 3: SORU YÜKLEME
    // ============================================================

    function soruYukle() {
        if (mevcutIdx >= sorular.length) {
            document.querySelector('.sahne').innerHTML =
                '<h2 style="color:#54a0ff;">🎪 SİRKİN YILDIZI!</h2>' +
                '<p>Tüm kesirleri doğru yerleştirdin!</p>';
            return;
        }
        const s = sorular[mevcutIdx];
        document.getElementById('kesir-goster').innerText = s.yazi;
        document.getElementById('mesaj').innerText = '';

        // Cambazı başlangıca al
        const cambaz = document.getElementById('cambaz');
        cambaz.style.left = '5%';           // Sola götür
        cambaz.classList.remove('sallan');   // Sallanmayı durdur
    }

    // ============================================================
    // 📚 BÖLÜM 4: BÖLGE SEÇİMİ
    // ============================================================

    function bolgeSec(secilen) {
        const s = sorular[mevcutIdx];
        const cambaz = document.getElementById('cambaz');

        // Seçilen bölgeye göre cambazı götür
        // (5% sol uç, 50% orta, 85% sağ uç)
        let hedefSol = '5%';
        if (secilen === 0)   hedefSol = '15%';   // Sola yakın
        if (secilen === 0.5) hedefSol = '50%';   // Ortaya
        if (secilen === 1)   hedefSol = '85%';   // Sağa yakın

        cambaz.style.left = hedefSol;     // CSS transition sayesinde animasyonlu hareket!
        cambaz.classList.add('sallan');   // Yürürken sallansın

        // setTimeout = 1 saniye bekle (animasyon bitsin), sonra kontrol et
        setTimeout(function() {
            cambaz.classList.remove('sallan');   // Sallanmayı durdur

            if (secilen === s.hedef) {
                // ✅ Doğru bölge!
                const mesaj = document.getElementById('mesaj');
                mesaj.innerText = '✅ ' + s.ipucu;
                mesaj.style.color = '#27ae60';
                setTimeout(function() { mevcutIdx++; soruYukle(); }, 2000);
            } else {
                // ❌ Yanlış bölge!
                alert('Dikkat! Cambaz dengesini kaybetti!\n\n💡 ' + s.ipucu);
                cambaz.style.left = '5%';   // Başa dön
            }
        }, 1000);
    }
    </script>

   , { yazi: "2/5", hedef: 0.5,
       ipucu: "5'in yarısı 2.5. 2 buna yakın, bu yüzden 1/2'ye yakın!" }
→ 2/5 = 0.4. 0.5'e uzaklığı 0.1, 0'a uzaklığı 0.4 → 1/2'ye daha yakın!

GÖREV 4 (Zor): Animasyon hızını değiştir
→ .cambaz { transition: left 1s ... } satırını bul
→ 1s yerine 2s yaz (2 saniye = yavaş yürüme)
→ setTimeout içindeki 1000'i de 2000 yap (2000ms = 2 saniye)
