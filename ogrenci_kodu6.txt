    <script>
    // ============================================================
    // 📚 BÖLÜM 1: GÖREV VERİLERİ
    // ============================================================

    // Her nesne bir analiz görevini temsil eder:
    //   konu = Hangi çevre konusu?
    //   soru = Ekrana gösterilecek soru metni
    //   tur  = "max" (büyük olanı seç) veya "min" (küçük olanı seç)
    //   a    = A bölgesinin bilgileri
    //     isim   = Bölge adı
    //     goster = Ekranda gösterilecek değer (kesir/ondalık/yüzde)
    //     deger  = Sayısal değer (karşılaştırmak için)
    //
    // 💡 max/min mantığı:
    //   GERİ DÖNÜŞÜM: Hangisi DAHA ÇOK yapıyor? → max (büyük)
    //   ORMAN KAYBI:  Hangisi DAHA AZ kaybediyor? → min (küçük)
    const gorevler = [
        {
            konu: "GERİ DÖNÜŞÜM ORANI",
            soru: "HANGİSİ DAHA ÇOK GERİ DÖNÜŞÜM YAPIYOR?",
            tur: "max",
            a: { isim: "ŞEHİR A", goster: "1/2",  deger: 0.5  },
            b: { isim: "ŞEHİR B", goster: "%25",   deger: 0.25 }
        },
        {
            konu: "ORMANLIK ALAN KAYBI",
            soru: "HANGİ BÖLGEDE ORMAN KAYBI DAHA AZ?",
            tur: "min",
            a: { isim: "BÖLGE X", goster: "0,4",   deger: 0.4 },
            b: { isim: "BÖLGE Y", goster: "3/10",  deger: 0.3 }
        },
        {
            konu: "YENİLENEBİLİR ENERJİ",
            soru: "HANGİSİ DAHA ÇOK GÜNEŞ ENERJİSİ KULLANIYOR?",
            tur: "max",
            a: { isim: "TESİS 1", goster: "19/20", deger: 0.95 },
            b: { isim: "TESİS 2", goster: "%90",   deger: 0.90 }
        }
    ];

    // ============================================================
    // 📚 BÖLÜM 2: DEĞİŞKENLER
    // ============================================================

    let mevcutIdx = 0;

    window.onload = function() { gorevYukle(); };

    // ============================================================
    // 📚 BÖLÜM 3: GÖREV YÜKLEME
    // ============================================================

    function gorevYukle() {
        if (mevcutIdx >= gorevler.length) {
            document.querySelector('.komuta').innerHTML =
                '<h2 style="color:#00ff80;text-align:center;">🌍 GEZEGEN KURTARILDI!</h2>' +
                '<p style="text-align:center;color:#aaa;">Tüm analizler tamamlandı!</p>';
            return;
        }

        const g = gorevler[mevcutIdx];
        document.getElementById('konu-goster').innerText = 'KONU: ' + g.konu;
        document.getElementById('soru').innerText        = g.soru;
        document.getElementById('mesaj').innerText       = '';

        // Her iki paneli güncelle
        panelGuncelle('a', g.a);
        panelGuncelle('b', g.b);
    }

    // ============================================================
    // 📚 BÖLÜM 4: PANEL GÜNCELLEME
    // ============================================================

    function panelGuncelle(id, bilgi) {
        document.getElementById('isim-'  + id).innerText = bilgi.isim;
        document.getElementById('deger-' + id).innerText = bilgi.goster;

        // Çubuk animasyonu:
        // Önce sıfır yap → Sonra setTimeout ile genişlet
        // Bu sayede her seferinde "dolan" animasyonu görünür
        const dolum = document.getElementById('dolum-' + id);
        dolum.style.width = '0%';   // Sıfırla

        // setTimeout(fonk, ms) = ms milisaniye sonra fonksiyonu çalıştır
        // 100ms = 0.1 saniye bekle (animasyon için küçük bir gecikme)
        setTimeout(function() {
            // bilgi.deger × 100 → Yüzdeye çevir
            dolum.style.width = (bilgi.deger * 100) + '%';
        }, 100);
    }

    // ============================================================
    // 📚 BÖLÜM 5: BÖLGE SEÇİMİ VE KONTROL
    // ============================================================

    function bolgeAc(secilen) {
        const g = gorevler[mevcutIdx];

        // Doğru cevabı bul:
        // tur="max" → Büyük olanı bul
        // tur="min" → Küçük olanı bul
        let dogru = '';
        if (g.tur === 'max') {
            dogru = (g.a.deger > g.b.deger) ? 'A' : 'B';
        } else {
            dogru = (g.a.deger < g.b.deger) ? 'A' : 'B';
        }

        const mesaj = document.getElementById('mesaj');
        if (secilen === dogru) {
            // ✅ Doğru bölge!
            mesaj.innerText  = '✅ Doğru! Eko-sistem dengelendi.';
            mesaj.style.color = '#00ff80';
            setTimeout(function() { mevcutIdx++; gorevYukle(); }, 2000);
        } else {
            const turIpucu = g.tur === 'max' ? 'Büyük olanı seç!' : 'Küçük olanı seç!';
            alert('Hatalı analiz!\n💡 ' + turIpucu + '\n' +
                  'A: ' + g.a.goster + '  |  B: ' + g.b.goster);
        }
    }
    </script>

→ const gorevler = [ ... ]; dizisine ekle:
   , {
       konu: "DOĞAL ALAN KORUMA",
       soru: "HANGİ BÖLGEDE DAHA FAZLA DOĞAL ALAN KORUNUYOR?",
       tur: "max",
       a: { isim: "İL A", goster: "3/4",  deger: 0.75 },
       b: { isim: "İL B", goster: "%60",  deger: 0.60 }
   }

GÖREV 3 (Orta): Çubuk animasyon süresini yavaşlat
→ .veri-dolum { transition: width 1s ease; } satırını bul
→ 1s yerine 2s yaz
→ setTimeout içindeki 100'ü 200 yap
→ Görsel efekt daha dramatik görünecek!

   const ilerleme = (mevcutIdx + 1) + '/' + gorevler.length;
