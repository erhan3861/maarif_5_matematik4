    <script>
    // ============================================================
    // 📚 BÖLÜM 1: SORU VERİLERİ
    // ============================================================

    // Her nesne bir soruyu temsil eder:
    //   sayi    = Ekranda gösterilen ondalık sayı
    //   deger   = Sayısal değer (nokta pozisyonu için)
    //   sol     = Sayı doğrusunun sol referans değeri
    //   sag     = Sayı doğrusunun sağ referans değeri
    //   dogru   = Doğru aralık cevabı (string)
    //   secenekler = 4 seçenek (biri doğru)
    //   ipucu   = Yanlış cevapta gösterilecek açıklama
    //
    // 💡 BASAMAK TUZAĞI:
    //   45,90 ≠ 45,09!!
    //   45,90 = 45 tam 90 yüzde = büyük
    //   45,09 = 45 tam 09 yüzde = küçük
    const sorular = [
        {
            sayi: "24,25", deger: 24.25,
            sol: 24.0, sag: 24.5,
            dogru: "24,0 – 24,5",
            secenekler: ["24,0 – 24,5", "24,5 – 25,0", "23,0 – 24,0", "25,0 – 26,0"],
            ipucu: "24,25 sayısı 24,0 ile 24,5 arasında yer alır."
        },
        {
            sayi: "45,09", deger: 45.09,
            sol: 45.0, sag: 45.1,
            dogru: "45,0 – 45,1",
            secenekler: ["45,9 – 46,0", "45,0 – 45,1", "45,5 – 45,9", "44,0 – 45,0"],
            ipucu: "TUZAK! 45,09 < 45,10. Yani 45,0 ile 45,1 arasındadır!"
        },
        {
            sayi: "11,99", deger: 11.99,
            sol: 11.9, sag: 12.0,
            dogru: "11,9 – 12,0",
            secenekler: ["11,0 – 11,9", "11,9 – 12,0", "12,0 – 12,1", "10,0 – 11,0"],
            ipucu: "11,99 tam 12,00'dan sadece 0,01 küçük!"
        }
    ];

    // ============================================================
    // 📚 BÖLÜM 2: DEĞİŞKENLER
    // ============================================================

    let mevcutIdx = 0;

    window.onload = function() { soruYukle(); };

    // ============================================================
    // 📚 BÖLÜM 3: SORU YÜKLEME
    // ============================================================

    function soruYukle() {
        if (mevcutIdx >= sorular.length) {
            document.querySelector('.mikroskop').innerHTML =
                '<h2 style="color:#00a8ff;">🔬 MİKROSKOP USTASI!</h2>' +
                '<p>Tüm ondalık sayıları doğru analiz ettin!</p>';
            return;
        }

        const s = sorular[mevcutIdx];
        document.getElementById('hedef-sayi').innerText = s.sayi;
        document.getElementById('sol-etiket').innerText = String(s.sol).replace('.', ',');
        document.getElementById('sag-etiket').innerText = String(s.sag).replace('.', ',');
        document.getElementById('mesaj').innerText = '';

        // Turuncu noktanın pozisyonu:
        // (deger - sol) / (sag - sol) → 0 ile 1 arasında bir oran
        // 5% + 90% × oran → Ekran yüzdesi
        const oran = (s.deger - s.sol) / (s.sag - s.sol);
        const yuzde = 5 + 90 * oran;
        document.getElementById('hedef-nokta').style.left = yuzde + '%';

        secenek_olustur(s);
    }

    // ============================================================
    // 📚 BÖLÜM 4: SEÇENEKLERİ OLUŞTUR
    // ============================================================

    function secenek_olustur(s) {
        const alan = document.getElementById('secenekAlan');
        alan.innerHTML = '';  // Önceki seçenekleri temizle

        // sort() ile seçenekleri karıştır: Math.random() - 0.5
        // Math.random() → 0-1 arası rastgele
        // - 0.5 → Bazen negatif bazen pozitif → Rastgele sıralama!
        const karisik = [...s.secenekler].sort(() => Math.random() - 0.5);

        // forEach = Dizinin her elemanı için fonksiyon çalıştır
        karisik.forEach(function(secim) {
            const div = document.createElement('div');
            div.className = 'secenek';
            div.innerText = secim;
            // Arrow function: (Kısa fonksiyon yazımı)
            div.onclick = () => cevapla(secim, s);
            alan.appendChild(div);
        });
    }

    // ============================================================
    // 📚 BÖLÜM 5: CEVAP KONTROLÜ
    // ============================================================

    function cevapla(secim, s) {
        const mesaj = document.getElementById('mesaj');
        if (secim === s.dogru) {
            // ✅ Doğru!
            mesaj.innerText = '✅ ' + s.ipucu;
            mesaj.style.color = '#2ecc71';
            setTimeout(function() { mevcutIdx++; soruYukle(); }, 2000);
        } else {
            // ❌ Yanlış
            alert('Yanlış aralık!\n\n💡 ' + s.ipucu);
        }
    }
