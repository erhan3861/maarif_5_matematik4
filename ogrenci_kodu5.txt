    <h2>🏷️ Etiket Dedektifi</h2>

    <div class="market">

            <span>REYON <span id="seviyeNo">1/3</span></span>

    <script>
    // ============================================================
    // 📚 BÖLÜM 1: ÜRÜN VERİLERİ
    // ============================================================

    // Her nesne bir "reyon" (seviye) verisidir:
    //   isim   = Ürün adı
    //   emoji  = Emoji ikon
    //   urunler = İki farklı marka ve fiyat
    //     id      = Benzersiz kimlik (hangisi seçildi anlamak için)
    //     fiyat   = Ondalık fiyat (TL)
    //     marka   = Marka adı
    //
    // 💡 TUZAK SORU: 45,90 vs 45,09
    //   45,90 = kırk beş tam doksan yüzde  → BÜYÜK
    //   45,09 = kırk beş tam dokuz yüzde   → KÜÇÜK
    //   Sondan sıfır okumayı YANILTIR!
    const reyonlar = [
        {
            isim: "Süt (1L)", emoji: "🥛",
            urunler: [
                { id: 1, fiyat: 24.50, marka: "Marka A" },
                { id: 2, fiyat: 24.25, marka: "Marka B" }  // UCUZ
            ]
        },
        {
            isim: "Pirinç (1kg)", emoji: "🍚",
            urunler: [
                { id: 1, fiyat: 45.90, marka: "Lüks" },    // TUZAK PAHALISI
                { id: 2, fiyat: 45.09, marka: "Eko" }      // UCUZ
            ]
        },
        {
            isim: "Çikolata", emoji: "🍫",
            urunler: [
                { id: 1, fiyat: 12.00, marka: "Tam" },
                { id: 2, fiyat: 11.99, marka: "Fırsat" }   // UCUZ (1 kuruş fark!)
            ]
        }
    ];

    // ============================================================
    // 📚 BÖLÜM 2: DEĞİŞKENLER
    // ============================================================

    let mevcutIdx   = 0;   // Hangi reyondayız?
    let toplamTasarruf = 0; // Toplam biriktirilen para (TL)

    window.onload = function() { reyonYukle(); };

    // ============================================================
    // 📚 BÖLÜM 3: REYON YÜKLEME
    // ============================================================

    function reyonYukle() {
        if (mevcutIdx >= reyonlar.length) {
            document.querySelector('.market').innerHTML =
                '<h2 style="color:#27ae60;text-align:center;">🛒 BİLİNÇLİ TÜKETİCİ!</h2>' +
                '<p style="text-align:center;">Toplam Tasarruf: <b>' +
                toplamTasarruf.toFixed(2).replace('.', ',') + ' TL</b></p>';
            return;
        }

        const r = reyonlar[mevcutIdx];
        document.getElementById('seviyeNo').innerText = (mevcutIdx + 1) + '/3';
        document.getElementById('mesaj').innerText    = '';

        const alan = document.getElementById('urunAlan');
        alan.innerHTML = '';  // Önceki ürünleri temizle

        // Ürünleri karıştır: Her seferinde farklı sıraya dizilsin
        // [...array] = Dizinin bir kopyasını oluşturur (orjinalini bozmaz)
        // .sort(() => Math.random() - 0.5) = Rastgele sırala
        let karisik = [...r.urunler].sort(() => Math.random() - 0.5);

        // forEach = Her ürün için kart oluştur
        karisik.forEach(function(urun) {
            const div = document.createElement('div');
            div.className = 'urun-karti';
            // onclick: Ok fonksiyonu (Arrow function)
            // () => tek satırlık fonksiyon demek
            div.onclick = () => fiyatKontrol(urun, karisik);

            // innerHTML ile kart içeriği oluştur (Template literal)
            div.innerHTML = `
                <span class="urun-emoji">${r.emoji}</span>
                <span class="urun-adi">${r.isim}
                    <small class="urun-marka">${urun.marka}</small>
                </span>
                <div class="fiyat-etiketi">${urun.fiyat.toFixed(2).replace('.', ',')}</div>
                <div class="barkod"></div>
            `;
            alan.appendChild(div);
        });
    }

    // ============================================================
    // 📚 BÖLÜM 4: FİYAT KONTROLÜ
    // ============================================================

    function fiyatKontrol(secilen, tumUrunler) {
        // reduce() = Diziyi tek bir değere indirgeme
        // prev.price < curr.price ? prev : curr
        // = Her karşılaştırmada daha ucuz olanı tut
        const enUcuz = tumUrunler.reduce(
            (prev, curr) => prev.fiyat < curr.fiyat ? prev : curr
        );

        const mesaj = document.getElementById('mesaj');

        if (secilen.id === enUcuz.id) {
            // ✅ En ucuzu seçtin!
            const pahali = tumUrunler.find(u => u.id !== secilen.id);
            const kazan  = pahali.fiyat - secilen.fiyat;
            toplamTasarruf += kazan;

            // toFixed(2) = 2 ondalık basamak göster
            document.getElementById('tasarruf').innerText =
                toplamTasarruf.toFixed(2).replace('.', ',');

            mesaj.innerText  = `✅ Harika! ${kazan.toFixed(2).replace('.', ',')} TL tasarruf ettin!`;
            mesaj.style.color = '#27ae60';

            // 2 saniye sonra bir sonraki reyona geç
            setTimeout(function() { mevcutIdx++; reyonYukle(); }, 2000);
        } else {
            alert('Daha uygunu var!\n' +
                  secilen.fiyat + ' > ' + enUcuz.fiyat +
                  '\n💡 Ondalık basamaklara dikkat!');
        }
    }
    </script>

GÖREV 1 (Kolay): Fiyat etiketinin rengini değiştir
→ .fiyat-etiketi { background: #fbc531; } satırını bul
→ #fbc531 (sarı) yerine #fd79a8 (pembe) yaz
→ Fiyat etiketleri pembe renk alacak!

GÖREV 2 (Kolay): Yeni bir ürün reyonu ekle
→ const reyonlar = [ ... ]; dizisine ekle:
   , {
     isim: "Su (1.5L)", emoji: "💧",
     urunler: [
         { id: 1, fiyat: 5.90, marka: "Dağ Suyu" },   // UCUZ
         { id: 2, fiyat: 5.99, marka: "Kaynak" }
     ]
   }
→ 5,90 < 5,99 → Dağ Suyu daha ucuz!

GÖREV 3 (Orta): Kartların sıralamasını KAPATMAK için
→ let karisik = [...r.urunler].sort(() => Math.random() - 0.5); satırını bul
→ Bunu şununla değiştir: let karisik = [...r.urunler];
→ Artık pahalı ürün her zaman solda görünür.
   Hangisi daha adil? Yorumunu ekle! // Sıralama kapatıldı

GÖREV 4 (Zor): Kaç tane doğru yapıldığını say
→ let dogru = 0; değişkenini BÖLÜM 2'ye ekle

   dogru++;

       'REYON ' + (mevcutIdx+1) + '/3 | ✅ ' + dogru + ' doğru';
